  задача 40
Найти решение системы линейных уравнений вида:

заданной своими коэффициентами A1, B1, C1, A2, B2, C2, если известно, что данная система имеет единственное решение.

Воспользоваться формулами:

x=(C1∗B2−C2∗B1)/D, y=(A1∗C2−A2∗C1)/D, где D=A1∗B2−A2∗B1.

#include <iostream>
using namespace std;
int main() {
    double A1, B1, C1, A2, B2, C2, D, x, y;    

    cout << "Запишіт коеф-ти першого рівняння (A1, B1, C1): ";
    cin >> A1 >> B1 >> C1;

    cout << "Запишіт коеф-ти другого рівняння (A2, B2, C2): ";
    cin >> A2 >> B2 >> C2;

    D=A1*B2-A2*B1;

  //!!!перевір що система має одне розв'язання!!!
    if (D == 0) {
        cout << "С-ма не має єдиного розв'язку" << std::endl;
    } else {
     
        x = (C1 * B2 - C2 * B1) / D;
        y = (A1 * C2 - A2 * C1) / D;

      
       cout << "Рез-т вирішення системи: x = " << x << ", y = " << y << endl;
    }

    return 0;
}

задача інт 24

Дни недели пронумерованы следующим образом: 0 — воскресенье, 1 — понедельник, 2 — вторник, …, 6 — суббота. Дано целое число K, лежащее в диапазоне 1–365. Определить номер дня недели для K-го дня года, если известно, что в этом году 1 января было понедельником.

#include <iostream>
using namespace std;
int main() {
    int K, day; 

    cout << "введіть номер дня року від 1 до 365: ";
    cin >> K;

    day = (K + 0) % 7;

    cout << "Номер дня тижня: " << day << endl;

    return 0;
}
